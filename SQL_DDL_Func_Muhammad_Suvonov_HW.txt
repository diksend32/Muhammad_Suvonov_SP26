3. inventory_in_stock    ---  checks whether that inventory item is available

```
CREATE OR REPLACE FUNCTION public.inventory_in_stock(
	p_inventory_id integer)
    RETURNS boolean
    LANGUAGE 'plpgsql'
    COST 100
    VOLATILE PARALLEL UNSAFE
AS $BODY$

DECLARE
    v_rentals INTEGER;
    v_out     INTEGER;
BEGIN
    -- AN ITEM IS IN-STOCK IF THERE ARE EITHER NO ROWS IN THE rental TABLE
    -- FOR THE ITEM OR ALL ROWS HAVE return_date POPULATED

    SELECT count(*) INTO v_rentals       
    FROM rental
    WHERE inventory_id = p_inventory_id;

    IF v_rentals = 0 THEN
      RETURN TRUE;
    END IF;                                      ---> if it is true it means that such copy is never rented

    SELECT COUNT(rental_id) INTO v_out
    FROM inventory LEFT JOIN rental USING(inventory_id)
    WHERE inventory.inventory_id = p_inventory_id
    AND rental.return_date IS NULL;            

    IF v_out > 0 THEN
      RETURN FALSE;
    ELSE
      RETURN TRUE;
    END IF;                               ---> checks if the inventory item is currently rented. If it is rented, it returns FALSE. If it is not rented, it returns TRUE.
END 
$BODY$;

ALTER FUNCTION public.inventory_in_stock(integer)
    OWNER TO postgres;
```
 
4   get_customer_balance --->  calculates how much money a customer currently owes or has as balance.

Replaced old implicit joins and messy structure with clean explicit joins to make the function easier to read and maintain.

```
CREATE OR REPLACE FUNCTION public.get_customer_balance(
    p_customer_id INTEGER,
    p_effective_date TIMESTAMPTZ
)
RETURNS NUMERIC
LANGUAGE plpgsql
AS
$$
DECLARE
    v_rentfees NUMERIC := 0;
    v_overfees NUMERIC := 0;
    v_payments NUMERIC := 0;
BEGIN

    
    SELECT COALESCE(SUM(f.rental_rate), 0)
    INTO v_rentfees
    FROM rental r
    JOIN inventory i ON i.inventory_id = r.inventory_id
    JOIN film f ON f.film_id = i.film_id
    WHERE r.customer_id = p_customer_id
      AND r.rental_date <= p_effective_date;


    SELECT COALESCE(SUM(
        CASE
            WHEN (r.return_date - r.rental_date) > (f.rental_duration * INTERVAL '1 day')
                THEN EXTRACT(DAY FROM (r.return_date - r.rental_date - (f.rental_duration * INTERVAL '1 day')))
            ELSE 0
        END
    ), 0)
    INTO v_overfees
    FROM rental r
    JOIN inventory i ON i.inventory_id = r.inventory_id
    JOIN film f ON f.film_id = i.film_id
    WHERE r.customer_id = p_customer_id
      AND r.rental_date <= p_effective_date;

    
    SELECT COALESCE(SUM(amount), 0)
    INTO v_payments
    FROM payment
    WHERE customer_id = p_customer_id
      AND payment_date <= p_effective_date;

    RETURN v_rentfees + v_overfees - v_payments;

END;
$$;
```

6. rewards_report -It produces a report of customers based on their rental activity and possibly their balance.
---  Replaced dynamic SQL + temp table with a direct query to make the function safer and simpler.
```
CREATE OR REPLACE FUNCTION public.rewards_report(
    min_monthly_purchases INTEGER,
    min_dollar_amount_purchased NUMERIC
)
RETURNS SETOF customer
LANGUAGE plpgsql
AS
$$
DECLARE
    start_date DATE;
    end_date DATE;
BEGIN

    -- sanity checks
    IF min_monthly_purchases <= 0 THEN
        RAISE EXCEPTION 'Minimum monthly purchases must be > 0';
    END IF;

    IF min_dollar_amount_purchased <= 0 THEN
        RAISE EXCEPTION 'Minimum dollar amount must be > 0';
    END IF;

    -- define reporting period (last 3rd month window like original logic)
    start_date := CURRENT_DATE - INTERVAL '3 months';
    end_date := CURRENT_DATE - INTERVAL '2 months';

    -- return matching customers directly (no temp table, no dynamic SQL)
    RETURN QUERY
    SELECT c.*
    FROM customer c
    JOIN payment p ON p.customer_id = c.customer_id
    WHERE p.payment_date BETWEEN start_date AND end_date
    GROUP BY c.customer_id
    HAVING COUNT(*) >= min_monthly_purchases
       AND SUM(p.amount) >= min_dollar_amount_purchased;

END;
$$;```
7.last_day   - it returns the last day of given month.

```
-- FUNCTION: public.last_day(timestamp with time zone)

-- DROP FUNCTION IF EXISTS public.last_day(timestamp with time zone);

CREATE OR REPLACE FUNCTION public.last_day(
	timestamp with time zone)
    RETURNS date
    LANGUAGE 'sql'
    COST 100
    IMMUTABLE STRICT PARALLEL UNSAFE
AS $BODY$

  SELECT CASE
    WHEN EXTRACT(MONTH FROM $1) = 12 THEN
      (((EXTRACT(YEAR FROM $1) + 1) operator(pg_catalog.||) '-01-01')::date - INTERVAL '1 day')::date
    ELSE
      ((EXTRACT(YEAR FROM $1) operator(pg_catalog.||) '-' operator(pg_catalog.||) (EXTRACT(MONTH FROM $1) + 1) operator(pg_catalog.||) '-01')::date - INTERVAL '1 day')::date
    END
$BODY$;

ALTER FUNCTION public.last_day(timestamp with time zone)
    OWNER TO postgres;
